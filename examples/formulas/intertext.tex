\begin{align}
  ax^2 + bx + c &= 0
  \intertext{因为 \(a \neq 0\), 两边同时除以\(a \), 得}
  x^2 + \cfrac{b}{a}x + \cfrac{c}{a}  &= 0
  \intertext{配方得}
  \left ( x + \cfrac{b}{2a} \right )^2 - \cfrac{b^2 - 4ac}{4a^2} &= 0
  \intertext{即}
  \left ( x + \cfrac{b}{2a} \right )^2 &= \cfrac{b^2 - 4ac}{4a^2}
\end{align}
